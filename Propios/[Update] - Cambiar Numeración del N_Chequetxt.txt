UPDATE SBA15
SET N_CHEQUE = 90031843 + (NumeroFila - 1)
FROM SBA15 C
JOIN (
    SELECT 
        N_CHEQUE,
        ROW_NUMBER() OVER (ORDER BY N_CHEQUE) AS NumeroFila
    FROM 
        SBA15
    WHERE 
        N_CHEQUE BETWEEN 90031779 AND 90031842 AND ULT_T_COMP = 'O/P'
) AS NC
ON C.N_CHEQUE = NC.N_CHEQUE
WHERE ULT_T_COMP = 'O/P'
